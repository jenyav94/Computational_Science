n=6

syms x;

pr_w(1)=-2*x;
pr_w(2)=2-6*x^2;
pr2_w(1)=-2;

for i=2:n-1
    ii=i-1;
%«десь i - номер коорд. ф-ии, ii - степень полинома якоби
    w(i+1)=((2*k+2*ii+1)*(k+ii+1)*x*w(i)-(k+ii)*(k+ii+1)*w(i-1))/((ii+1)*(2*k+ii+1));
    pr_w(i+1)=diff(w(i+1),x);
end
